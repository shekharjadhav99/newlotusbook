function save_commentary(val=''){
	if(val==''){
		val = $(#Commentary).val();
	}
	if(val!=''){
		 jQuery.ajax({ 
			
			url 'httpnewlotusbook.comoddsSettingSavecommentary'+val,			
			success function(data) {
			type post,
            url 'httpnewlotusbook.comoddsSettingSave',
           datadatastring,only input
			data {commentary 'commentary', val val},
            cache false,
            success function(output)
            {   	
				new PNotify({
						  title 'Success',
						  text 'Saved successfully',
						  type 'success',
						  styling 'bootstrap3',
						  delay500
					  });
					  
			},
			error function (request) {
				new PNotify({
						  title 'Error',
						  text 'Not Saved successfully',
						  type 'error',
						  styling 'bootstrap3',
						  delay500
					  });
				
			},
		}); 
		
	}
	
}
